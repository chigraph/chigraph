195  if (err != 0) {

202  err = git_remote_fetch(origin, nullptr, &opts, nullptr);
203  if (err != 0) {

205  return res;

217 
218  return 0;
219 
220  }, &oid_to_merge );

225  if (err != 0) {

258  if (err != 0) {

268  return res;
269  }
270 
271

273  git_oid oid;
274  err = git_index_write_tree_to(&oid, head, repo);
275  if (err != 0) {

289

293  if (err != 0) {

295  return res;
296  }

303  }

308  git_signature* committerSignature;
309  err = git_signature_now(&committerSignature, "Chigraph Fetch", "russellgreene8@gmail.com");
310  if (err != 0) {
311  res.addEntry("EUKN", "Failed to create git signature", {{"Repo Path", repoPath.string()}, {"Error Message", giterr_last()->message}});
312  return res;

316  git_commit* origin_master_commit;

318  if (err != 0) {

320  return res;

328  return res;
329  }

335  return res;
336  }

346

352  }
353  }
354

357

361 
362  if (type == VCSType::Unknown) {
363  res.addEntry("EUKN", "Could not resolve URL for module", {{"Module Name", name.string()}});
364  return res;
365  }
366  Expects(type == VCSType::Git);
367

370 
371  // clone it

373  int err = git_clone(&repo, url.c_str(), absCloneInto.string().c_str(), nullptr);
374 
375  // check for error
376  if (err != 0) {
377  res.addEntry("EUKN", "Failed to clone repository", {{"Error Code", err}, {"Error Message", giterr_last()->message}});
378  return res;
379  }
380  }
381

384  // TODO: is there a cleaner way to do this?
385  nlohmann::json j;
386  try {
387  fs::ifstream file{fileName};
388  file >> j;
389  } catch (std::exception& e) {

391  }

400  }
401 
402  }
403 
404  return res;
