55 void NodeInstance::setType(std::unique_ptr<NodeType> newType) {
56  // delete exec connections that are out of range
57  // start at one past the end
58  for (size_t id = newType->execInputs().size(); id < inputExecConnections.size(); ++id) {
59  while (!inputExecConnections[id].empty()) {
60  Expects(inputExecConnections[id][0].first); // should never fail...
61  disconnectExec(*inputExecConnections[id][0].first, inputExecConnections[id][0].second);
62  }
63  }

67  auto& conn = outputExecConnections[id];
68  if (conn.first != nullptr) { disconnectExec(*this, id); }
69  }

103 Result connectData(NodeInstance& lhs, size_t lhsConnID, NodeInstance& rhs, size_t rhsConnID) {
104  Result res = {};
105 
106  if (&lhs.function() != &rhs.function()) {
107  res.addEntry("EUKN", "Cannot connect two nodes of different graphs",
108  {}); // TODO: better errors
109  return res;
110  }
111 
112  // make sure the connection exists
113  // the input to the connection is the output to the node
114  if (lhsConnID >= lhs.outputDataConnections.size()) {
115  auto dataOutputs = nlohmann::json::array();
116  for (auto& output : lhs.type().dataOutputs()) {
117  dataOutputs.push_back({{output.name, output.type.qualifiedName()}});
118  }
119 
120  res.addEntry("E22", "Output Data connection doesn't exist in node",
121  {{"Requested ID", lhsConnID},
122  {"Node Type", lhs.type().qualifiedName()},
123  {"Node JSON", rhs.type().toJSON()},
124  {"Node Output Data Connections", dataOutputs}});
125  }
126  if (rhsConnID >= rhs.inputDataConnections.size()) {
127  auto dataInputs = nlohmann::json::array();
128  for (auto& output : rhs.type().dataInputs()) {
129  dataInputs.push_back({{output.name, output.type.qualifiedName()}});
130  }
131 
132  res.addEntry("E23", "Input Data connection doesn't exist in node",
133  {{"Requested ID", rhsConnID},
134  {"Node Type", rhs.type().qualifiedName()},
135  {"Node JSON", rhs.type().toJSON()},
136  {"Node Input Data Connections", dataInputs}});
137  }
138 
139  // if there are errors, back out
140  if (!res) { return res; }
141  // make sure the connection is of the right type
142  if (lhs.type().dataOutputs()[lhsConnID].type != rhs.type().dataInputs()[rhsConnID].type) {
143  res.addEntry("E24", "Connecting data nodes with different types is invalid",
144  {{"Left Hand Type", lhs.type().dataOutputs()[lhsConnID].type.qualifiedName()},
145  {"Right Hand Type", rhs.type().dataInputs()[rhsConnID].type.qualifiedName()},
146  {"Left Node JSON", rhs.type().toJSON()},
147  {"Right Node JSON", rhs.type().toJSON()}});
148  return res;
149  }
150 
151  // if we are replacing a connection, disconnect it
152  if (rhs.inputDataConnections[rhsConnID].first != nullptr) {
153  res += disconnectData(lhs, lhsConnID, rhs);
154  if (!res) { return res; }
155  }
156 
157  lhs.outputDataConnections[lhsConnID].emplace_back(&rhs, rhsConnID);
158  rhs.inputDataConnections[rhsConnID] = {&lhs, lhsConnID};
