56 void NodeInstance::setType(std::unique_ptr<NodeType> newType) {
57  // delete exec connections that are out of range
58  // start at one past the end
59  for (size_t id = newType->execInputs().size(); id < inputExecConnections.size(); ++id) {
60  while (!inputExecConnections[id].empty()) {
61  Expects(inputExecConnections[id][0].first); // should never fail...
62  disconnectExec(*inputExecConnections[id][0].first, inputExecConnections[id][0].second);
63  }
64  }

66

68  auto& conn = outputExecConnections[id];
69  if (conn.first != nullptr) { disconnectExec(*this, id); }
70  }
71  outputExecConnections.resize(newType->execOutputs().size(), std::make_pair(nullptr, ~0));
72

76  !(type().dataInputs()[id].type == newType->dataInputs()[id].type)) {
77  disconnectData(*conn.first, conn.second, *this);
78  }
79  ++id;
80  }

82

85  // keep the connections if they're still good
86  if (newType->dataOutputs().size() > id &&

105  Result res = {};
106 
107  if (&lhs.function() != &rhs.function()) {
108  res.addEntry("EUKN", "Cannot connect two nodes of different graphs",
109  {}); // TODO: better errors
110  return res;
111  }
112 
113  // make sure the connection exists
114  // the input to the connection is the output to the node
115  if (lhsConnID >= lhs.outputDataConnections.size()) {
116  auto dataOutputs = nlohmann::json::array();
117  for (auto& output : lhs.type().dataOutputs()) {
118  dataOutputs.push_back({{output.name, output.type.qualifiedName()}});
119  }
120 
121  res.addEntry("E22", "Output Data connection doesn't exist in node",
122  {{"Requested ID", lhsConnID},
123  {"Node Type", lhs.type().qualifiedName()},
124  {"Node JSON", rhs.type().toJSON()},
125  {"Node Output Data Connections", dataOutputs}});
126  }
127  if (rhsConnID >= rhs.inputDataConnections.size()) {
128  auto dataInputs = nlohmann::json::array();
129  for (auto& output : rhs.type().dataInputs()) {
130  dataInputs.push_back({{output.name, output.type.qualifiedName()}});
131  }
132 
133  res.addEntry("E23", "Input Data connection doesn't exist in node",
134  {{"Requested ID", rhsConnID},
135  {"Node Type", rhs.type().qualifiedName()},
136  {"Node JSON", rhs.type().toJSON()},
137  {"Node Input Data Connections", dataInputs}});
138  }
139 
140  // if there are errors, back out
141  if (!res) { return res; }
142  // make sure the connection is of the right type
143  if (lhs.type().dataOutputs()[lhsConnID].type != rhs.type().dataInputs()[rhsConnID].type) {
144  res.addEntry("E24", "Connecting data nodes with different types is invalid",
145  {{"Left Hand Type", lhs.type().dataOutputs()[lhsConnID].type.qualifiedName()},
146  {"Right Hand Type", rhs.type().dataInputs()[rhsConnID].type.qualifiedName()},
147  {"Left Node JSON", rhs.type().toJSON()},
148  {"Right Node JSON", rhs.type().toJSON()}});
149  return res;
150  }
151 
152  // if we are replacing a connection, disconnect it
153  if (rhs.inputDataConnections[rhsConnID].first != nullptr) {
154  res += disconnectData(lhs, lhsConnID, rhs);
155  if (!res) { return res; }
156  }
157

159  rhs.inputDataConnections[rhsConnID] = {&lhs, lhsConnID};
