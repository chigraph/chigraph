606  const fs::path& CFile = direntry;
607 
608  if (!fs::is_regular_file(CFile) ||

610  CFile.extension() == ".cpp" || CFile.extension() == ".cxx" ||

612  continue;
613  }
614

617  compileCCode(llvm::sys::fs::getMainExecutable(nullptr, nullptr).c_str(), "",

761 Result GraphModule::nodeTypeFromName(boost::string_view name, const nlohmann::json& jsonData,
762  std::unique_ptr<NodeType>* toFill) {
763  Result res = {};
764 
765  // see if it's a C call
766  if (cEnabled() && name == "c-call") {
767  if (!jsonData.is_object()) {
768  res.addEntry("WUKN", "Data for c-call must be an object", {{"Given Data"}, jsonData});
769  }
770 
771  std::string code;
772  if (jsonData.is_object() && jsonData.find("code") != jsonData.end() &&
773  jsonData["code"].is_string()) {
774  code = jsonData["code"];
775  } else {
776  res.addEntry(
777  "WUKN",
778  "Data for c-call must have a pair with the key of code and that the data is a "
779  "string",
780  {{"Given Data"}, jsonData});
781  }
782 
783  std::string function;
784  if (jsonData.is_object() && jsonData.find("function") != jsonData.end() &&
785  jsonData["function"].is_string()) {
786  function = jsonData["function"];
787  } else {
788  res.addEntry(
789  "WUKN",
790  "Data for c-call must have a pair with the key of function and that the data is a "
791  "string",
792  {{"Given Data"}, jsonData});
793  }
794 
795  std::vector<std::string> extraFlags;
796  if (jsonData.is_object() && jsonData.find("extraflags") != jsonData.end() &&
797  jsonData["extraflags"].is_array()) {
798  for (const auto& flag : jsonData["extraflags"]) {
799  std::string str = flag;
800  extraFlags.emplace_back(std::move(str));
801  }
802  } else {
803  res.addEntry("WUKN", "Data for c-call must have an extraflags array",
804  {{"Given Data"}, jsonData});
805  }
806 
807  std::vector<NamedDataType> inputs;
808  if (jsonData.is_object() && jsonData.find("inputs") != jsonData.end() &&
809  jsonData["inputs"].is_array()) {
810  inputs.reserve(jsonData["inputs"].size());
811  for (const auto& jsonArg : jsonData["inputs"]) {
812  std::string docstring, qualifiedName;
813  std::tie(docstring, qualifiedName) = parseObjectPair(jsonArg);
814 
815  std::string moduleName, typeName;
816  std::tie(moduleName, typeName) = parseColonPair(qualifiedName);
817 
818  DataType ty;
819  res += context().typeFromModule(moduleName, typeName, &ty);
820 
821  inputs.emplace_back(docstring, ty);
822  }
823 
824  } else {
825  res.addEntry("WUKN", "Data for c-call must have an inputs array",
826  {{"Given Data"}, jsonData});
827  }
828 
829  DataType output;
830  if (jsonData.is_object() && jsonData.find("output") != jsonData.end()) {
831  auto& outputJson = jsonData["output"];
832 
833  // if it's a string, there's a return type
834  if (outputJson.is_string()) {
835  std::string moduleName, typeName;
836  std::tie(moduleName, typeName) = parseColonPair(outputJson);
837 
838  res += context().typeFromModule(moduleName, typeName, &output);
839  }
840  // now if it isn't null, then there was an error
841  else if (!outputJson.is_null()) {
842  res.addEntry("WUKN",
843  R"("output" element in c-call must be either null or a string)",
844  {{"Given Data"}, jsonData});
845  }
846  } else {
847  res.addEntry(
848  "WUKN",
849  "Data for c-call must have an output element that is either null or a string",
850  {{"Given Data"}, jsonData});
851  }
852 
853  *toFill = std::make_unique<CFuncNode>(*this, code, function, extraFlags, inputs, output);
854  return res;
855 
856  } else if (name == "c-call") {
857  res.addEntry("EUKN", "Tried to use a c-call node without c support enabled in the module",
858  {});
859  return res;
860  }
861 
862  auto graph = functionFromName(name);
863 
864  if (graph == nullptr) {
865  // if it wasn't found, then see if it's a struct breaker or maker
866  std::string nameStr{name};
867  if (nameStr.substr(0, 6) == "_make_") {
868  auto str = structFromName(nameStr.substr(6));
869  if (str != nullptr) {
870  *toFill = std::make_unique<MakeStructNodeType>(*str);
871  return res;
872  }
873  }
874  if (nameStr.substr(0, 7) == "_break_") {
875  auto str = structFromName(nameStr.substr(7));
876  if (str != nullptr) {
877  *toFill = std::make_unique<BreakStructNodeType>(*str);
878  return res;
879  }
880  }
881  if (nameStr.substr(0, 5) == "_get_") {
882  if (jsonData.is_string()) {
883  std::string module, typeName;
884 
885  std::tie(module, typeName) = parseColonPair(jsonData);
886 
887  DataType ty;
888  res += context().typeFromModule(module, typeName, &ty);
889 
890  *toFill =
891  std::make_unique<GetLocalNodeType>(*this, NamedDataType{nameStr.substr(5), ty});
892  } else {
893  res.addEntry("EUKN", "Json data for _get_ node type isn't a string",
894  {{"Given Data", jsonData}});
895  }
896  return res;
897  }
898  if (nameStr.substr(0, 5) == "_set_") {
899  if (jsonData.is_string()) {
900  std::string module, typeName;
901 
902  std::tie(module, typeName) = parseColonPair(jsonData);
903 
904  DataType ty;
905  res += context().typeFromModule(module, typeName, &ty);
906 
907  *toFill = std::make_unique<SetLocalNodeType>(
908  *this, NamedDataType{nameStr.substr(5), std::move(ty)});
909  } else {
910  res.addEntry("EUKN", "Json data for _set_ node type isn't a string",
911  {{"Given Data", jsonData}});
912  }
913  return res;
914  }
915 
916  // if we get here than it's for sure not a thing
917  res.addEntry("EUKN", "Graph not found in module",
918  {{"Module Name", fullName()}, {"Requested Graph", name.to_string()}});
919  return res;
920  }
921 
922  *toFill = std::make_unique<GraphFuncCallType>(*this, name.to_string(), &res);
923  return res;
924 }
