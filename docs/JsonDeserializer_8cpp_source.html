13  gsl::cstring_span<> fullName, GraphModule** toFill) {

15 
16  // create the module

18  if (toFill != nullptr) { *toFill = createdModule; }
19 
20  // load dependencies
21  {
22  auto iter = input.find("dependencies");
23  if (iter == input.end()) {
24  res.addEntry("E38", "No dependencies element in module", {});
25  return res;
26  }
27  if (!iter->is_array()) {
28  res.addEntry("E39", "dependencies element isn't an array", {});
29  return res;
30  }
31 
32  for (const auto& dep : *iter) {
33  if (!dep.is_string()) {
34  res.addEntry("E40", "dependency isn't a string", {{"Actual Data", dep}});
35  continue;
36  }
37  res += createdModule->addDependency(dep);
38 
39  if (!res) { return res; }
40  }
41  }
42 
43  // load types
44  {
45  auto iter = input.find("types");
46  if (iter == input.end() || !iter->is_object()) {
47  res.addEntry("EUKN", "No types object in module", {{}});
48  return res;
49  }
50 
51  // declare them
52  for (auto tyIter = iter->begin(); tyIter != iter->end(); ++tyIter) {
53  createdModule->getOrCreateStruct(tyIter.key());
54  }
55  // load them
56  for (auto tyIter = iter->begin(); tyIter != iter->end(); ++tyIter) {
57  res += jsonToGraphStruct(*createdModule, tyIter.key(), tyIter.value());
58  }
59  }
60 
61  // load graphs
62  {
63  auto iter = input.find("graphs");
64  if (iter == input.end()) {
65  res.addEntry("E41", "no graphs element in module", {});
66  return res;
67  }
68  if (!iter->is_array()) {
69  res.addEntry("E42", "graph element isn't an array", {{"Actual Data", *iter}});
70  return res;
71  }
72 
73  std::vector<GraphFunction*> functions;
74  functions.resize(iter->size());
75 
76  // create forward declarations
77  auto id = 0ull;
78  for (const auto& graph : *iter) {
79  res += createGraphFunctionDeclarationFromJson(*createdModule, graph, &functions[id]);
80 
81  ++id;
82  }
83 
84  if (!res) { return res; }
85 
86  // load the graphs
87  id = 0;
88  for (const auto& graph : *iter) {
89  res += jsonToGraphFunction(*functions[id], graph);
90 
91  ++id;
92  }
93  }
94

98 Result createGraphFunctionDeclarationFromJson(GraphModule& createInside,
99  const nlohmann::json& input, GraphFunction** toFill) {
100  Result res;
101 
102  if (!input.is_object()) {
103  res.addEntry("E1", "Graph json isn't a JSON object", {});
104  return res;
105  }
106  // make sure it has a type element
107  if (input.find("type") == input.end()) {
108  res.addEntry("E2", R"(JSON in graph doesn't have a "type" element)", {});
109  return res;
110  }
111  if (input["type"] != "function") {
112  res.addEntry("E3", "JSON in graph doesn't have a function type", {});
113  return res;
114  }
115  // make sure there is a name
116  if (input.find("name") == input.end()) {
117  res.addEntry("E4", "JSON in graph doesn't have a name parameter", {});
118  return res;
119  }
120  std::string name = input["name"];
121 
122  if (input.find("data_inputs") == input.end() || !input["data_inputs"].is_array()) {
123  res.addEntry("E43", "JSON in graph doesn't have an data_inputs array", {});
124  return res;
125  }
126 
127  std::vector<NamedDataType> datainputs;
128  for (auto param : input["data_inputs"]) {
129  std::string docString, qualifiedType;
130  std::tie(docString, qualifiedType) = parseObjectPair(param);
131 
132  std::string moduleName, name;
133  std::tie(moduleName, name) = parseColonPair(qualifiedType);
134 
135  DataType ty;
136  res += createInside.context().typeFromModule(moduleName, name, &ty);
137 
138  if (!res) { return res; }
139 
140  datainputs.emplace_back(docString, ty);
141  }
142 
143  if (input.find("data_outputs") == input.end() || !input["data_outputs"].is_array()) {
144  res.addEntry("E44", "JSON in graph doesn't have an data_outputs array", {});
145  return res;
146  }
147 
148  std::vector<NamedDataType> dataoutputs;
149  for (auto param : input["data_outputs"]) {
150  std::string docString, qualifiedType;
151  std::tie(docString, qualifiedType) = parseObjectPair(param);
152 
153  std::string moduleName, name;
154  std::tie(moduleName, name) = parseColonPair(qualifiedType);
155 
156  DataType ty;
157  res += createInside.context().typeFromModule(moduleName, name, &ty);
158 
159  if (!res) { return res; }
160 
161  dataoutputs.emplace_back(docString, ty);
162  }
163 
164  // get exec I/O
165  if (input.find("exec_inputs") == input.end() || !input["exec_inputs"].is_array()) {
166  res.addEntry("EUKN", "JSON in graph doesn't have an exec_inputs array", {});
167  return res;
168  }
169 
170  std::vector<std::string> execinputs;
171  for (const auto& param : input["exec_inputs"]) {
172  std::string name = param;
173 
174  execinputs.emplace_back(name);
175  }
176 
177  if (input.find("exec_outputs") == input.end() || !input["exec_outputs"].is_array()) {
178  res.addEntry("EUKN", "JSON in graph doesn't have an data_outputs array", {});
179  return res;
180  }
181 
182  std::vector<std::string> execoutputs;
183  for (const auto& param : input["exec_outputs"]) {
184  std::string name = param;
185 
186  execoutputs.emplace_back(name);
187  }
188 
189  // construct it
190  auto created =
191  createInside.getOrCreateFunction(name, datainputs, dataoutputs, execinputs, execoutputs);
192  if (toFill != nullptr) { *toFill = created; }
193 
194  return res;

197 Result jsonToGraphFunction(GraphFunction& createInside, const nlohmann::json& input) {
198  Result res;
199 
200  // read the nodes
201  if (input.find("nodes") == input.end() || !input["nodes"].is_object()) {
202  res.addEntry("E5", "JSON in graph doesn't have nodes object", {});
203  return res;
204  }
205 
206  for (auto nodeiter = input["nodes"].begin(); nodeiter != input["nodes"].end(); ++nodeiter) {
207  auto node = nodeiter.value();
208  std::string nodeid = nodeiter.key();
209  if (node.find("type") == node.end() || !node.find("type")->is_string()) {
210  res.addEntry("E6", R"(Node doesn't have a "type" string)", {{"nodeid", nodeid}});
211  return res;
212  }
213  std::string fullType = node["type"];
214  std::string moduleName, typeName;
215  std::tie(moduleName, typeName) = parseColonPair(fullType);
216 
217  if (moduleName.empty() || typeName.empty()) {
218  res.addEntry("E7", "Incorrect qualified module name (should be module:type)",
219  {{"nodeid", nodeid}, {"Requested Qualified Name", fullType}});
220  return res;
221  }
222 
223  if (node.find("data") == node.end()) {
224  res.addEntry("E9", "Node doens't have a data section", {"nodeid", nodeid});
225  return res;
226  }
227 
228  std::unique_ptr<NodeType> nodeType;
229  res += createInside.context().nodeTypeFromModule(moduleName, typeName, node["data"],
230  &nodeType);
231  if (!res) { continue; }
232 
233  auto testIter = node.find("location");
234  if (testIter == node.end()) {
235  res.addEntry("E12", "Node doesn't have a location.", {{"nodeid", nodeid}});
236  continue;
237  }
238 
239  // make sure it is the right size
240  if (!testIter.value().is_array()) {
241  res.addEntry("E10", "Node doesn't have a location that is an array.",
242  {{"nodeid", nodeid}});
243  continue;
244  }
245 
246  if (testIter.value().size() != 2) {
247  res.addEntry("E11", "Node doesn't have a location that is an array of size 2.",
248  {{"nodeid", nodeid}});
249  continue;
250  }
251 
252  createInside.insertNode(std::move(nodeType), node["location"][0], node["location"][1],
253  nodeid);
254  }
255 
256  size_t connID = 0;
257  // read the connections
258  {
259  auto connIter = input.find("connections");
260  if (connIter == input.end() || !connIter->is_array()) {
261  res.addEntry("E13", "No connections array in function", {});
262  return res;
263  }
264  for (auto& connection : input["connections"]) {
265  if (connection.find("type") == connection.end() ||
266  !connection.find("type")->is_string()) {
267  res.addEntry("E14", "No type string in connection", {"connectionid", connID});
268  continue;
269  }
270  std::string type = connection["type"];
271  bool isData = type == "data";
272  // it either has to be "input" or "exec"
273  if (!isData && type != "exec") {
274  res.addEntry("E15", "Unrecognized connection type",
275  {{"connectionid", connID}, {"Found Type", type}});
276  continue;
277  }
278 
279  if (connection.find("input") == connection.end()) {
280  res.addEntry("E16", "No input element in connection", {{"connectionid", connID}});
281  continue;
282  }
283  if (!connection.find("input")->is_array() || connection.find("input")->size() != 2 ||
284  !connection.find("input")->operator[](0).is_string() ||
285  !connection.find("input")->operator[](1).is_number_integer()) {
286  res.addEntry(
287  "E17",
288  "Incorrect connection input format, must be an array of of a string (node id) "
289  "and int (connection id)",
290  {{"connectionid", connID}, {"Requested Type", *connection.find("input")}});
291  continue;
292  }
293  std::string InputNodeID = connection["input"][0];
294  int InputConnectionID = connection["input"][1];
295 
296  if (connection.find("output") == connection.end()) {
297  res.addEntry("E18", "No output element in connection", {{"connectionid", connID}});
298  continue;
299  }
300  if (!connection.find("output")->is_array() || connection.find("output")->size() != 2 ||
301  !connection.find("output")->operator[](0).is_string() ||
302  !connection.find("output")->operator[](1).is_number_integer()) {
303  res.addEntry(
304  "E19",
305  "Incorrect connection output format, must be an array of a string (node id) "
306  "and int (connection id)",
307  {{"connectionid", connID}, {"Requested Type", *connection.find("output")}});
308  continue;
309  }
310  std::string OutputNodeID = connection["output"][0];
311  int OutputConnectionID = connection["output"][1];
312 
313  // make sure the nodes exist
314  if (createInside.nodes().find(InputNodeID) == createInside.nodes().end()) {
315  res.addEntry("E20", "Input node for connection doesn't exist",
316  {{"connectionid", connID}, {"Requested Node", InputNodeID}});
317  continue;
318  }
319  if (createInside.nodes().find(OutputNodeID) == createInside.nodes().end()) {
320  res.addEntry("E21", "Output node for connection doesn't exist",
321  {{"connectionid", connID}, {"Requested Node", OutputNodeID}});
322  continue;
323  }
324 
325  // connect
326  // these functions do bounds checking, it's okay
327  if (isData) {
328  res += connectData(*createInside.nodes()[InputNodeID], InputConnectionID,
329  *createInside.nodes()[OutputNodeID], OutputConnectionID);
330  } else {
331  res += connectExec(*createInside.nodes()[InputNodeID], InputConnectionID,
332  *createInside.nodes()[OutputNodeID], OutputConnectionID);
333  }
334 
335  ++connID;
336  }
337  }
338  return res;
339 }

341 Result jsonToGraphStruct(GraphModule& mod, gsl::cstring_span<> name, const nlohmann::json& input,
342  GraphStruct** toFill) {
343  Result res;
344 
345  if (!input.is_array()) {
346  res.addEntry("EUKN", "Graph Struct json has to be an array", {{"Given JSON", input}});
347  return res;
348  }
349 
350  auto createdStruct = mod.getOrCreateStruct(gsl::to_string(name));
351  if (toFill != nullptr) { *toFill = createdStruct; }
352 
353  for (const auto& str : input) {
354  if (!str.is_object()) {
355  res.addEntry("EUKN", "Graph Struct entry must be an object", {{"Given JSON", str}});
356  continue;
357  }
358 
359  if (str.size() != 1) {
360  res.addEntry("EUKN", "Graph Struct entry must have size of 1", {{"Size", str.size()}});
361  continue;
362  }
363 
364  std::string docstring, qualifiedType;
365  std::tie(docstring, qualifiedType) = parseObjectPair(str);
366 
367  // parse the type
368  std::string typeModuleName, typeName;
369  std::tie(typeModuleName, typeName) = parseColonPair(qualifiedType);
370 
371  DataType ty;
372  res += mod.context().typeFromModule(typeModuleName, typeName, &ty);
373 
374  if (!res) { continue; }
375 
376  createdStruct->addType(ty, docstring, createdStruct->types().size());
377  }
378 
379  return res;
380 }

382 std::pair<std::string, std::string> parseObjectPair(const nlohmann::json& object) {
383  if (!object.is_object()) { return {}; }
384 
385  auto iter = object.begin();
386  if (iter == object.end()) { return {}; }
387 
388  std::string key = iter.key();
389  std::string val = iter.value();
390 
391  // make sure it's the only element
392  ++iter;
393  if (iter != object.end()) { return {}; }
394 
395  return {key, val};
396 }
