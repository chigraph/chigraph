13  const boost::filesystem::path& fullName, GraphModule** toFill) {

15

18  if (toFill != nullptr) { *toFill = createdModule; }
19 
20  // load dependencies
21  {
22  auto iter = input.find("dependencies");
23  if (iter == input.end()) {
24  res.addEntry("E38", "No dependencies element in module", {});
25  return res;
26  }
27  if (!iter->is_array()) {
28  res.addEntry("E39", "dependencies element isn't an array", {});
29  return res;
30  }
31 
32  for (const auto& dep : *iter) {

41  }

46  if (iter == input.end() || !iter->is_object()) {
47  res.addEntry("EUKN", "No types object in module", {{}});

49  }
50

60

64  if (iter == input.end()) {
65  res.addEntry("E41", "no graphs element in module", {});
66  return res;
67  }
68  if (!iter->is_array()) {
69  res.addEntry("E42", "graph element isn't an array", {{"Actual Data", *iter}});

71  }
72

74  functions.resize(iter->size());
75

80 
81  ++id;
82  }
83

85

87  id = 0;
88  for (const auto& graph : *iter) {

100  Result res;
101 
102  if (!input.is_object()) {
103  res.addEntry("E1", "Graph json isn't a JSON object", {});
104  return res;
105  }
106  // make sure it has a type element
107  if (input.find("type") == input.end() || !input["type"].is_string()) {
108  res.addEntry("E2", R"(JSON in graph doesn't have a "type" element)",
109  {{"Module Name", createInside.fullName()}});
110  return res;
111  }
112  if (input["type"] != "function") {
113  res.addEntry("E3", "JSON in graph doesn't have a function type",
114  {{"Module Name", createInside.fullName()}});
115  return res;
116  }
117  // make sure there is a name
118  if (input.find("name") == input.end() || !input["name"].is_string()) {
119  res.addEntry("E4", "JSON in graph doesn't have a name parameter",
120  {{"Module Name", createInside.fullName()}});
121  return res;
122  }
123  std::string name = input["name"];
124 
125  // load the description
126  if (input.find("description") == input.end() || !input["description"].is_string()) {
127  res.addEntry("E50", "JSON in graph doesn't have a description string",
128  {{"Function Name", name}, {"Module Name", createInside.fullName()}});
129  return res;
130  }
131  std::string description = input["description"];
132 
133  if (input.find("data_inputs") == input.end() || !input["data_inputs"].is_array()) {
134  res.addEntry("E43", "JSON in graph doesn't have an data_inputs array", {});
135  return res;
136  }
137 
138  std::vector<NamedDataType> datainputs;
139  for (auto param : input["data_inputs"]) {
140  std::string docString, qualifiedType;

142 
143  std::string moduleName, name;
144  std::tie(moduleName, name) = parseColonPair(qualifiedType);
145 
146  DataType ty;
147  res += createInside.context().typeFromModule(moduleName, name, &ty);
148 
149  if (!res) { return res; }
150 
151  datainputs.emplace_back(docString, ty);
152  }
153 
154  if (input.find("data_outputs") == input.end() || !input["data_outputs"].is_array()) {
155  res.addEntry("E44", "JSON in graph doesn't have an data_outputs array", {});
156  return res;
157  }
158

160  for (auto param : input["data_outputs"]) {
161  std::string docString, qualifiedType;

166 
167  DataType ty;
168  res += createInside.context().typeFromModule(moduleName, name, &ty);
169 
170  if (!res) { return res; }
171 
172  dataoutputs.emplace_back(docString, ty);
173  }
174 
175  // get exec I/O
176  if (input.find("exec_inputs") == input.end() || !input["exec_inputs"].is_array()) {
177  res.addEntry("E48", "JSON in graph doesn't have an exec_inputs array", {});

180

182  for (const auto& param : input["exec_inputs"]) {

191  }

195  std::string name = param;
196

209 Result jsonToGraphFunction(GraphFunction& createInside, const nlohmann::json& input) {
210  Result res;
211 
212  // read the local variables
213  if (input.find("local_variables") == input.end() || !input["local_variables"].is_object()) {
214  res.addEntry("E45", "JSON in graph doesn't have a local_variables object", {});
215 
216  return res;
217  }
218 
219  for (auto localiter = input["local_variables"].begin();
220  localiter != input["local_variables"].end(); ++localiter) {
221  std::string localName = localiter.key();
222 
223  if (!localiter.value().is_string()) {
224  res.addEntry("E46", "Local variable vaue in json wasn't a string",
225  {{"Given local variable json", localiter.value()}});
226 
227  continue;
228  }
229 
230  // parse the type names
231  std::string qualifiedType = localiter.value();
232 
233  std::string moduleName, typeName;
234  std::tie(moduleName, typeName) = parseColonPair(qualifiedType);
235 
236  DataType ty;
237  res += createInside.context().typeFromModule(moduleName, typeName, &ty);
238 
239  if (!res) { continue; }
240 
241  createInside.getOrCreateLocalVariable(localName, ty);
242  }
243 
244  // read the nodes
245  if (input.find("nodes") == input.end() || !input["nodes"].is_object()) {
246  res.addEntry("E5", "JSON in graph doesn't have nodes object", {});
247  return res;
248  }
249 
250  for (auto nodeiter = input["nodes"].begin(); nodeiter != input["nodes"].end(); ++nodeiter) {
251  auto node = nodeiter.value();
252  std::string nodeid = nodeiter.key();
253  if (node.find("type") == node.end() || !node.find("type")->is_string()) {
254  res.addEntry("E6", R"(Node doesn't have a "type" string)", {{"nodeid", nodeid}});
255  return res;
256  }
257  std::string fullType = node["type"];
258  std::string moduleName, typeName;
259  std::tie(moduleName, typeName) = parseColonPair(fullType);
260 
261  if (moduleName.empty() || typeName.empty()) {
262  res.addEntry("E7", "Incorrect qualified module name (should be module:type)",
263  {{"nodeid", nodeid}, {"Requested Qualified Name", fullType}});
264  return res;
265  }
266 
267  if (node.find("data") == node.end()) {
268  res.addEntry("E9", "Node doens't have a data section", {"nodeid", nodeid});
269  return res;
270  }
271 
272  std::unique_ptr<NodeType> nodeType;
273  res += createInside.context().nodeTypeFromModule(moduleName, typeName, node["data"],
274  &nodeType);
275  if (!res) { continue; }
276 
277  auto testIter = node.find("location");
278  if (testIter == node.end()) {
279  res.addEntry("E12", "Node doesn't have a location.", {{"nodeid", nodeid}});
280  continue;
281  }
282 
283  // make sure it is the right size
284  if (!testIter.value().is_array()) {
285  res.addEntry("E10", "Node doesn't have a location that is an array.",
286  {{"nodeid", nodeid}});
287  continue;
288  }
289 
290  if (testIter.value().size() != 2) {
291  res.addEntry("E11", "Node doesn't have a location that is an array of size 2.",
292  {{"nodeid", nodeid}});
293  continue;
294  }
295 
296  try {
297  auto uuidNodeID = boost::uuids::string_generator()(nodeid);
298 
299  createInside.insertNode(std::move(nodeType), node["location"][0], node["location"][1],
300  uuidNodeID);
301  } catch (std::exception& e) {
302  res.addEntry("E51", "Invalid UUID string", {{"string", nodeid}});
303  }
304  }
305 
306  size_t connID = 0;
307  // read the connections
308  {
309  auto connIter = input.find("connections");
310  if (connIter == input.end() || !connIter->is_array()) {
311  res.addEntry("E13", "No connections array in function", {});
312  return res;
313  }
314  for (auto& connection : input["connections"]) {
315  if (connection.find("type") == connection.end() ||
316  !connection.find("type")->is_string()) {
317  res.addEntry("E14", "No type string in connection", {"connectionid", connID});
318  continue;
319  }
320  std::string type = connection["type"];
321  bool isData = type == "data";
322  // it either has to be "input" or "exec"
323  if (!isData && type != "exec") {
324  res.addEntry("E15", "Unrecognized connection type",
325  {{"connectionid", connID}, {"Found Type", type}});
326  continue;
327  }
328 
329  if (connection.find("input") == connection.end()) {
330  res.addEntry("E16", "No input element in connection", {{"connectionid", connID}});
331  continue;
332  }
333  if (!connection.find("input")->is_array() || connection.find("input")->size() != 2 ||
334  !connection.find("input")->operator[](0).is_string() ||
335  !connection.find("input")->operator[](1).is_number_integer()) {
336  res.addEntry(
337  "E17",
338  "Incorrect connection input format, must be an array of of a string (node id) "
339  "and int (connection id)",
340  {{"connectionid", connID}, {"Requested Type", *connection.find("input")}});
341  continue;
342  }
343  std::string InputNodeID = connection["input"][0];
344 
345  boost::uuids::uuid InputNodeIDUUID;
346  try {
347  InputNodeIDUUID = boost::uuids::string_generator()(InputNodeID);
348  } catch (std::exception&) {
349  res.addEntry("EUKN", "Invalid UUID string in connection",
350  {{"string", InputNodeID}});
351 
352  return res;
353  }
354  int InputConnectionID = connection["input"][1];
355 
356  if (connection.find("output") == connection.end()) {
357  res.addEntry("E18", "No output element in connection", {{"connectionid", connID}});
358  continue;
359  }
360  if (!connection.find("output")->is_array() || connection.find("output")->size() != 2 ||
361  !connection.find("output")->operator[](0).is_string() ||
362  !connection.find("output")->operator[](1).is_number_integer()) {
363  res.addEntry(
364  "E19",
365  "Incorrect connection output format, must be an array of a string (node id) "
366  "and int (connection id)",
367  {{"connectionid", connID}, {"Requested Type", *connection.find("output")}});
368  continue;
369  }
370  std::string OutputNodeID = connection["output"][0];
371 
372  boost::uuids::uuid OutputNodeIDUUID;
373  try {
374  OutputNodeIDUUID = boost::uuids::string_generator()(OutputNodeID);
375  } catch (std::exception&) {
376  res.addEntry("EUKN", "Invalid UUID string in connection",
377  {{"string", OutputNodeID}});
378 
379  return res;
380  }
381  int OutputConnectionID = connection["output"][1];
382 
383  // make sure the nodes exist
384  if (createInside.nodes().find(InputNodeIDUUID) == createInside.nodes().end()) {
385  res.addEntry("E20", "Input node for connection doesn't exist",
386  {{"connectionid", connID}, {"Requested Node", InputNodeID}});
387  continue;
388  }
389  if (createInside.nodes().find(OutputNodeIDUUID) == createInside.nodes().end()) {
390  res.addEntry("E21", "Output node for connection doesn't exist",
391  {{"connectionid", connID}, {"Requested Node", OutputNodeID}});
392  continue;
393  }
394 
395  // connect
396  // these functions do bounds checking, it's okay
397  if (isData) {
398  res += connectData(*createInside.nodes()[InputNodeIDUUID], InputConnectionID,
399  *createInside.nodes()[OutputNodeIDUUID], OutputConnectionID);
400  } else {
401  res += connectExec(*createInside.nodes()[InputNodeIDUUID], InputConnectionID,
402  *createInside.nodes()[OutputNodeIDUUID], OutputConnectionID);
403  }
404 
405  ++connID;
406  }
407  }
408  return res;
409 }

412  GraphStruct** toFill) {
413  Result res;
414 
415  if (!input.is_array()) {

417  return res;
418  }
419 
420  auto createdStruct = mod.getOrCreateStruct(name.to_string());
421  if (toFill != nullptr) { *toFill = createdStruct; }
422 
423  for (const auto& str : input) {

426  continue;

452 std::pair<std::string, std::string> parseObjectPair(const nlohmann::json& object) {
453  if (!object.is_object()) { return {}; }
454 
455  auto iter = object.begin();
456  if (iter == object.end()) { return {}; }
457 
458  std::string key = iter.key();
459  std::string val = iter.value();
460 
461  // make sure it's the only element
462  ++iter;
463  if (iter != object.end()) { return {}; }
464 
465  return {key, val};
466 }
