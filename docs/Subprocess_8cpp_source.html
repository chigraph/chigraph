23  // Get the error message, if any.
24  DWORD errorMessageID = GetLastError();
25  if (errorMessageID == 0) return std::string(); // No error message has been recorded
26 
27  LPSTR messageBuffer = nullptr;

123  secAttributes.bInheritHandle = true;
124  secAttributes.lpSecurityDescriptor = nullptr;
125 
126  HANDLE stdoutWrite;
127  HANDLE stderrWrite;
128  HANDLE stdinRead;
129

131  if (!CreatePipe(&mPimpl->StdOut_Read, &stdoutWrite, &secAttributes, 0)) {
132  res.addEntry("EUKN", "Failed to create stdout pipe",
133  {{"Error Message", GetLastErrorAsString()}});
134  return res;
135  }
136  // make sure the read handle isn't inhereited
137  if (!SetHandleInformation(mPimpl->StdOut_Read, HANDLE_FLAG_INHERIT, 0)) {
138  res.addEntry("EUKN", "Failed to SetHandleInformation",
139  {{"Error Message", GetLastErrorAsString()}});
140  return res;
141  }
142 
143  // create stderr pipe
144  if (!CreatePipe(&mPimpl->StdErr_Read, &stderrWrite, &secAttributes, 0)) {
145  res.addEntry("EUKN", "Failed to create stderr pipe",
146  {{"Error Message", GetLastErrorAsString()}});
147  return res;
148  }
149  // make sure the read handle isn't inhereited
150  if (!SetHandleInformation(mPimpl->StdErr_Read, HANDLE_FLAG_INHERIT, 0)) {
151  res.addEntry("EUKN", "Failed to SetHandleInformation",
152  {{"Error Message", GetLastErrorAsString()}});
153  return res;
154  }
155 
156  // create stdin pipe
157  if (!CreatePipe(&stdinRead, &mPimpl->StdIn_Write, &secAttributes, 0)) {
158  res.addEntry("EUKN", "Failed to create stdin pipe",
159  {{"Error Message", GetLastErrorAsString()}});
160  return res;
161  }
162  // make sure the read handle isn't inhereited
163  if (!SetHandleInformation(mPimpl->StdIn_Write, HANDLE_FLAG_INHERIT, 0)) {
164  res.addEntry("EUKN", "Failed to SetHandleInformation",
165  {{"Error Message", GetLastErrorAsString()}});
166  return res;
167  }
168 
169  // finally start the process
171 
172  // create commandline
173  std::wstring commandLine = L'"' + mExePath.wstring() + L'"';
174

176  std::wstring wCmd;
177  std::copy(cmd.begin(), cmd.end(), std::back_inserter(wCmd));
178 
179  boost::algorithm::replace_all(wCmd, L"\"", L"\\\"");
180 
181  commandLine += L" \"" + wCmd + L'"';
182  }
183 
184  ZeroMemory(&mPimpl->procInfo, sizeof(PROCESS_INFORMATION));
185 
186  // setup startupInfo
187  STARTUPINFO startupInfo;
188  ZeroMemory(&startupInfo, sizeof(STARTUPINFO));
189

191  startupInfo.hStdError = stderrWrite;
192  startupInfo.hStdOutput = stdoutWrite;
193  startupInfo.hStdInput = stdinRead;
194  startupInfo.dwFlags |= STARTF_USESTDHANDLES;
195

197  if (!CreateProcessW(nullptr, &commandLine[0], nullptr, nullptr, TRUE, 0, nullptr,

205  CloseHandle(mPimpl->procInfo.hThread);

208  CloseHandle(stderrWrite);

210  CloseHandle(stdinRead);

368 Result Subprocess::start() {
369  assert(!started() && "Cannot start a Subprocess twice");
370 
371  Result res;
372 
373  // create pipes
374  if (pipe(mPimpl->stdinPipe.data()) != 0) {
375  res.addEntry("EUKN", "Failed to create stdin pipe", {{"Error message", strerror(errno)}});
376  return res;
377  }
378  if (pipe(mPimpl->stdoutPipe.data()) != 0) {
379  res.addEntry("EUKN", "Failed to create stdout pipe", {{"Error message", strerror(errno)}});
380  return res;
381  }
382  if (pipe(mPimpl->stderrPipe.data()) != 0) {
383  res.addEntry("EUKN", "Failed to create stderr pipe", {{"Error message", strerror(errno)}});
384  return res;
385  }
386 
387  // fork!
388  mPimpl->childPID = fork();
389 
390  if (mPimpl->childPID == -1) {
391  res.addEntry("EUKN", "Failed to fork", {{"Error message", strerror(errno)}});
392  return res;
393  }
394 
395  // child process
396  if (mPimpl->childPID == 0) {
397  boost::filesystem::current_path(mWorkingDir);
398 
399  // make read end of stdin pipe the stdin stream, and same for the other pipes
400  dup2(mPimpl->stdinPipe[0], 0);
401  dup2(mPimpl->stdoutPipe[1], 1);
402  dup2(mPimpl->stdoutPipe[1], 2);
403 
404  // close the other fds, we don't need them
405  close(mPimpl->stdinPipe[0]);
406  close(mPimpl->stdinPipe[1]);
407  close(mPimpl->stdoutPipe[0]);
408  close(mPimpl->stdinPipe[1]);
409  close(mPimpl->stderrPipe[0]);
410  close(mPimpl->stdinPipe[1]);
411 
412  // close open fds for the process (other than 0, 1, and 2 which are the std streams)
413  // https://stackoverflow.com/questions/899038/getting-the-highest-allocated-file-descriptor/899533#899533
414  int fd_max = static_cast<int>(sysconf(_SC_OPEN_MAX)); // truncation is safe
415  for (int fd = 3; fd < fd_max; fd++) close(fd);
416 
417  // set group
418  setpgid(0, 0);
419 
420  // make argv
421  std::vector<char*> argv;
422  std::string exePathStr = mExePath.string();
423  argv.push_back(&exePathStr[0]);
424  for (auto& arg : mArguments) { argv.push_back(&arg[0]); }
425 
426  // run the process
427  execvp(mExePath.c_str(), argv.data());
428 
429  _exit(EXIT_FAILURE);
430  }
431 
432  // parent process
433 
434  // close the ends of the pipes we don't use
435  close(mPimpl->stdinPipe[0]);
436  mPimpl->stdinPipe[0] = -1;
437  close(mPimpl->stdoutPipe[1]);
438  mPimpl->stdoutPipe[1] = -1;
439  close(mPimpl->stderrPipe[1]);
440  mPimpl->stderrPipe[1] = -1;
441 
442  // start threads
443  mPimpl->stdoutThread = std::thread([this] {
444 
445  std::array<char, 4096> buffer;
446  while (true) {
447  // read from stdout
448  ssize_t bytesRead = read(mPimpl->stdoutPipe[0], buffer.data(), buffer.size());
449 
450  // error
451  if (bytesRead == -1) {
452  return;
453  }
454 
455  // 0 is EOF
456  if (bytesRead != 0 && mStdOutHandler) { mStdOutHandler(buffer.data(), bytesRead); }
457  }
458  });
459 
460  mPimpl->stderrThread = std::thread([this] {
461 
462  std::array<char, 4096> buffer;
463  while (true) {
464  // read from stdout
465  ssize_t bytesRead = read(mPimpl->stderrPipe[0], buffer.data(), buffer.size());
466 
467  // error
468  if (bytesRead == -1) {
469  return;
470  }
471 
472  // 0 is EOF
473  if (bytesRead != 0 && mStdErrHandler) { mStdErrHandler(buffer.data(), bytesRead); }
474  }
475  });
476 
477  mStarted = true;
478 
479  return {};
480 }
