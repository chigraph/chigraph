185 // Codegens a single input to a node
186 void codegenHelper(NodeInstance* node, unsigned execInputID, llvm::BasicBlock* block,
187  llvm::BasicBlock* allocblock, llvm::Module* mod, llvm::Function* f,
188  std::unordered_map<NodeInstance*, cache>& nodeCache, Result& res)
189 {
190  llvm::IRBuilder<> builder(block);
191 
192  // get inputs and outputs
193  std::vector<llvm::Value*> io;
194 
195  // add inputs
196  {
197  size_t inputID = 0;
198  for (auto& param : node->inputDataConnections) {
199  // make sure everything is A-OK
200  if (param.first == nullptr) {
201  res.addEntry("EUKN", "No data input to node",
202  {{"nodeid", node->id()}, {"input ID", inputID},
203  {"nodetype", node->type().qualifiedName()}
204  });
205 
206  return;
207  }
208 
209  auto cacheiter = nodeCache.find(param.first);
210 
211  if (cacheiter == nodeCache.end()) {
212  res.addEntry("EUKN", "Failed to find in cache", {{"nodeid", param.first->id()}});
213 
214  return;
215  }
216 
217  auto& cacheObject = cacheiter->second;
218  if (param.second >= cacheObject.outputs.size()) {
219  res.addEntry("EUKN", "No data input to node",
220  {{"nodeid", node->id()}, {"input ID", inputID},
221  {"nodetype", node->type().qualifiedName()}
222  });
223 
224  return;
225  }
226 
227  // get pointers to the objects
228  auto value = cacheObject.outputs[param.second];
229  // dereference
230  io.push_back(builder.CreateLoad(
231  value, node->type().dataInputs()[inputID].second)); // TODO: pass ptr to value
232 
233  // make sure it's the right type
234  if (io[io.size() - 1]->getType() !=
235  node->type().dataInputs()[inputID].first.llvmType()) {
236  res.addEntry("EINT", "Internal codegen error: unexpected type in cache.",
237  {{"Expected LLVM type",
238  stringifyLLVMType(node->type().dataInputs()[inputID].first.llvmType())},
239  {"Found type", stringifyLLVMType(io[io.size() - 1]->getType())},
240  {"Node ID", node->id()}, {"Input ID", inputID}});
241  }
242 
243  ++inputID;
244  }
245  }
246 
247  // get outputs
248  {
249  llvm::IRBuilder<> allocBuilder(allocblock);
250 
251  // create outputs
252  auto& outputCache = nodeCache[node].outputs;
253 
254  for (auto& output : node->type().dataOutputs()) {
255  // TODO: research address spaces
256  llvm::AllocaInst* alloc =
257  allocBuilder.CreateAlloca(output.first.llvmType(), nullptr, output.second);
258  alloc->setName(output.second);
259  outputCache.push_back(alloc);
260  io.push_back(alloc);
261 
262  // make sure the type is right
263  if (llvm::PointerType::get(output.first.llvmType(), 0) != alloc->getType()) {
264  res.addEntry("EINT",
265  "Internal codegen error: unexpected type returned from alloca.",
266  {{"Expected LLVM type",
267  stringifyLLVMType(llvm::PointerType::get(output.first.llvmType(), 0))},
268  {"Yielded type", stringifyLLVMType(alloc->getType())},
269  {"Node ID", node->id()}});
270  }
271  }
272  }
273 
274  // create output blocks
275  std::vector<llvm::BasicBlock*> outputBlocks;
276  std::vector<llvm::BasicBlock*> unusedBlocks;
277  for (auto idx = 0ull; idx < node->outputExecConnections.size(); ++idx) {
278  auto outBlock =
279  llvm::BasicBlock::Create(f->getContext(), node->type().execOutputs()[idx], f);
280  outputBlocks.push_back(outBlock);
281  if (node->outputExecConnections[idx].first != nullptr) {
282  outBlock->setName("node_" + node->outputExecConnections[idx].first->id());
283  } else {
284  unusedBlocks.push_back(outBlock);
285  }
286  }
287 
288  // codegen
289  res += node->type().codegen(execInputID, mod, f, io, block, outputBlocks);
290  if (!res) {
291  return;
292  }
293 
294  // TODO: sequence nodes
295  for (auto& bb : unusedBlocks) {
296  llvm::IRBuilder<> builder(bb);
297 
298  builder.CreateRet(
299  llvm::ConstantInt::get(llvm::IntegerType::getInt32Ty(mod->getContext()), 0, true));
300  }
301 
302  // recurse!
303  for (auto idx = 0ull; idx < node->outputExecConnections.size(); ++idx) {
304  auto& output = node->outputExecConnections[idx];
305  if (output.first != nullptr) {
306  codegenHelper(
307  output.first, output.second, outputBlocks[idx], allocblock, mod, f, nodeCache, res);
308  }
309  }
310 }

312 Result GraphFunction::compile(llvm::Module* mod, llvm::Function** ret_func) const
313 {
314  Result res;
315 
316  auto entry = entryNode();
317  if (entry == nullptr) {
318  res.addEntry("EUKN", "No entry node", {});
319  return res;
320  }
321 
322  // make sure that the entry node has the functiontype
323  if (!std::equal(
324  dataInputs().begin(), dataInputs().end(), entry->type().dataOutputs().begin())) {
325  nlohmann::json inFunc = nlohmann::json::array();
326  for (auto& in : dataInputs()) {
327  inFunc.push_back({{in.second, in.first.qualifiedName()}});
328  }
329 
330  nlohmann::json inEntry = nlohmann::json::array();
331  for (auto& in :
332  entry->type().dataOutputs()) { // outputs to entry are inputs to the function
333  inEntry.push_back({{in.second, in.first.qualifiedName()}});
334  }
335 
336  res.addEntry("EUKN", "Inputs to function doesn't match function inputs",
337  {{"Function Inputs", inFunc}, {"Entry Inputs", inEntry}});
338  return res;
339  }
340 
341  // make sure that the entry node has the functiontype
342  if (!std::equal(dataOutputs().begin(), dataOutputs().end(), dataOutputs().begin())) {
343  nlohmann::json outFunc = nlohmann::json::array();
344  for (auto& out : dataOutputs()) {
345  outFunc.push_back({{out.second, out.first.qualifiedName()}});
346  }
347 
348  nlohmann::json outEntry = nlohmann::json::array();
349  for (auto& out : dataOutputs()) {
350  // inputs to the exit are outputs to the function
351  outEntry.push_back({{out.second, out.first.qualifiedName()}});
352  }
353 
354  res.addEntry("EUKN", "Outputs to function doesn't match function exit",
355  {{"Function Outputs", outFunc}, {"Entry Outputs", outEntry}});
356  return res;
357  }
358 
359  llvm::Function* f = llvm::cast<llvm::Function>(mod->getOrInsertFunction(
360  mangleFunctionName(module().fullName(), name()), functionType())); // TODO: name mangling
361  llvm::BasicBlock* allocblock = llvm::BasicBlock::Create(mod->getContext(), "alloc", f);
362  llvm::BasicBlock* block = llvm::BasicBlock::Create(mod->getContext(), name() + "_entry", f);
363  auto blockcpy = block;
364 
365  // follow "linked list"
366  unsigned execInputID =
367  entry->outputExecConnections[0].second; // the exec connection to codegen from
368 
369  std::unordered_map<NodeInstance*, cache> nodeCache;
370 
371  // set argument names
372  auto idx = 0ull;
373  for (auto& arg : f->getArgumentList()) {
374  // the first one is the input exec ID
375  if (idx == 0) {
376  arg.setName("inputexec_id");
377  ++idx;
378  continue;
379  }
380  // all the - 1's is becaues the first is the inputexec_id
381  if (idx - 1 < entry->type().dataOutputs().size()) {
382  // dataOutputs to entry are inputs to the function
383  arg.setName(entry->type()
384  .dataOutputs()[idx - 1]
385  .second); // it starts with inputs, which are outputs to entry
386  } else {
387  arg.setName(dataOutputs()[idx - 1 - entry->type().dataOutputs().size()]
388  .second); // then the outputs, which are inputs to exit
389  }
390  ++idx;
391  }
392 
393  codegenHelper(entry, execInputID, block, allocblock, mod, f, nodeCache, res);
394 
395  llvm::IRBuilder<> allocbuilder(allocblock);
396  allocbuilder.CreateBr(blockcpy);
397 
398  *ret_func = f;
399  return res;
400 }

437 Result GraphFunction::removeNode(NodeInstance* nodeToRemove)
438 {
439  Result res;
440 
441  // disconnect it's connections
442 
443  // disconnect input exec
444  for (const auto& execSlot : nodeToRemove->inputExecConnections) {
445  for (const auto& pair : execSlot) {
446  if (pair.first) {
447  res += disconnectExec(*pair.first, pair.second);
448  }
449  }
450  }
451  // disconnect output exec
452  auto ID = 0ull;
453  for (const auto& pair : nodeToRemove->outputExecConnections) {
454  if (pair.first) {
455  res += disconnectExec(*nodeToRemove, ID);
456  }
457  ++ID;
458  }
459 
460  // disconnect input data
461  for (const auto& pair : nodeToRemove->inputDataConnections) {
462  if (pair.first) {
463  res += disconnectData(*pair.first, pair.second, *nodeToRemove);
464  }
465  }
466 
467  // disconnect output data
468  ID = 0ull;
469  for (const auto& dataSlot : nodeToRemove->outputDataConnections) {
470  for (const auto& pair : dataSlot) {
471  if (pair.first) {
472  disconnectData(*nodeToRemove, ID, *pair.first);
473  }
474  }
475  ++ID;
476  }
477  // then delete the node
478  graph().nodes().erase(nodeToRemove->id());
479 
480  return res;
481 }

557 Result GraphFunction::validateGraph() const
558 {
559  Result res;
560  // make sure all connections connect back
561 
562  for (const auto& node : graph().nodes()) {
563  // go through input data
564  auto id = 0ull;
565  for (const auto& conn : node.second->inputDataConnections) {
566  // make sure it connects back
567  bool connectsBack = false;
568  for (const auto& remoteConn : conn.first->outputDataConnections[conn.second]) {
569  if (remoteConn.first == node.second.get() && remoteConn.second == id) {
570  connectsBack = true;
571  break;
572  }
573  }
574  if (!connectsBack) {
575  res.addEntry("EUKN", "Data connection doesn't connect back",
576  {{"Left Node", conn.first->id()}, {"Right Node", node.second->id()},
577  {"Right input ID", id}});
578  }
579  ++id;
580  }
581 
582  // output data
583  id = 0ull;
584  for (const auto& outputDataSlot : node.second->outputDataConnections) {
585  // this connection type can make multiple connections, so two for loops are needed
586  for (const auto& connection : outputDataSlot) {
587  auto& remoteConn = connection.first->inputDataConnections[connection.second];
588  if (remoteConn.first != node.second.get() || remoteConn.second != id) {
589  res.addEntry("EUKN", "Data connection doesn't connect back",
590  {{"Left Node", node.second->id()}, {"Right Node", connection.first->id()},
591  {"Right input ID", connection.second}});
592  }
593  }
594  ++id;
595  }
596 
597  // input exec
598  id = 0ull;
599  for (const auto& inputExecSlot : node.second->inputExecConnections) {
600  // this connection type can make multiple connections, so two for loops are needed
601  for (const auto& connection : inputExecSlot) {
602  auto& remoteConn = connection.first->outputExecConnections[connection.second];
603  if (remoteConn.first != node.second.get() || remoteConn.second != id) {
604  res.addEntry("EUKN", "Exec connection doesn't connect back",
605  {{"Left Node", connection.first->id()}, {"Right Node", node.second->id()},
606  {"Left output ID", connection.second}}); // TODO: better diagnostics
607  }
608  }
609  ++id;
610  }
611 
612  // output exec
613  id = 0ull;
614  for (const auto& connection : node.second->outputExecConnections) {
615  bool connectsBack = false;
616  for (const auto& remoteConnection :
617  connection.first->inputExecConnections[connection.second]) {
618  if (remoteConnection.second == id && remoteConnection.first == node.second.get()) {
619  connectsBack = true;
620  break;
621  }
622  }
623 
624  if (!connectsBack) {
625  res.addEntry("EUKN", "Exec connection doesn't connect back",
626  {{"Left Node", node.second->id()}, {"Right Node", connection.first->id()},
627  {"Left output ID", id}});
628  }
629 
630  ++id;
631  }
632  }
633 
634  return res;
635 }
